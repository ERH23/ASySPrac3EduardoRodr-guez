        FuncionEditFieldLabel          matlab.ui.control.Label

        onePanelWidth = 576;

    % Callbacks that handle component events
    methods (Access = private)

        % Value changing function: FuncionEditField
        function FuncionEditFieldValueChanging(app, event)


        end

        % Value changed function: LmitesuperiorEditField
        function LmitesuperiorEditFieldValueChanged(app, event)

        end

        function LmiteinferiorEditFieldValueChanged(app, event)

        end

        % Callback function: not associated with a component
        function UIAxesButtonDown(app, event)

        end

        % Button pushed function: TransformacinVerticalButton
        function TransformacinVerticalButtonPushed(app, event)
            funcStr = app.FuncionEditField.Value;
            liminf = str2double(app.LmiteinferiorEditField.Value);
            limsup = str2double(app.LmitesuperiorEditField.Value);
            factvert = str2double(app.FactordetransformacionverticalEditField);
            func = str2func(['@(x)', funcStr]);
            x = linspace(liminf,limsup,1000);
            y = func(x)*factvert;
            plot(app.UIAxes2,x,y);

        end

        % Button pushed function: MostrarButton
        function MostrarButtonPushed(app, event)
            funcStr = app.FuncionEditField.Value;
            Liminf = str2double(app.LmiteinferiorEditField.Value);
            Limsup = str2double(app.LmitesuperiorEditField.Value);
            func = str2func(['@(x)', funcStr]);
            x = linspace(Liminf,Limsup,1000);
            y = func(x);
            plot(app.UIAxes,x,y);
        end

        % Button down function: UIAxes
        function UIAxesButtonDown2(app, event)

        end

        % Button pushed function: TransformacinHorizontalButton
        function TransformacinHorizontalButtonPushed(app, event)
            Funcion = app.FuncionEditField.Value;
            Liminf = str2double(app.LmiteinferiorEditField.Value);
            Limsup = str2double(app.LmitesuperiorEditField.Value);
            Factor = str2double(app.FactordetransformacionhorizontalEditField.Value);

            func = str2func(['@(x)' Funcion]);
            x = linspace(Liminf,Limsup,1000);
            y = func(x / Factor);
            plot(app.UIAxes3,x,y);

        end

        % Changes arrangement of the app based on UIFigure width
        function updateAppLayout(app, event)
            currentFigureWidth = app.UIFigure.Position(3);
            if(currentFigureWidth <= app.onePanelWidth)
                % Change to a 2x1 grid
                app.GridLayout.RowHeight = {583, 583};
                app.GridLayout.ColumnWidth = {'1x'};
                app.RightPanel.Layout.Row = 2;
                app.RightPanel.Layout.Column = 1;
            else
                % Change to a 1x2 grid
                app.GridLayout.RowHeight = {'1x'};
                app.GridLayout.ColumnWidth = {265, '1x'};
                app.RightPanel.Layout.Row = 1;
                app.RightPanel.Layout.Column = 2;
            end
        end
    end

    % Component initialization
    methods (Access = private)

        % Create UIFigure and components
        function createComponents(app)

            % Create UIFigure and hide until all components are created
            app.UIFigure = uifigure('Visible', 'off');
            app.UIFigure.AutoResizeChildren = 'off';
            app.UIFigure.Position = [100 100 716 583];

            app.UIFigure.SizeChangedFcn = createCallbackFcn(app, @updateAppLayout, true);

            % Create GridLayout
            app.GridLayout = uigridlayout(app.UIFigure);
            app.GridLayout.ColumnWidth = {265, '1x'};
            app.GridLayout.RowHeight = {'1x'};
